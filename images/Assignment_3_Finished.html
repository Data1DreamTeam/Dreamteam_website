# --------------------------------------- CORRELATION COEFFICIENT ----------------------------------------
# The correlation coefficient gives us an indication on how much two variables are linearly correlated 
# one to the other. It is defined as the covariance of the two variables divided by the product of their standard deviations.
# In formula:
# R(x,y) = ( COV (x,y) / sd(x)*sd(y) )

#importing numpy:
import numpy as np

#reading the file and calculating the correlation coefficient with .corrcoef 
df_2 = pd.read_csv("/Users/pietro/Documents/UNI/MADE/Data1/lab-02/data/cleaned-goodreads.csv")
corr_coeff = np.corrcoef(df_2['review_count'], df_2['rating'])[0, 1]

#printing
print("-------------------------------------")
print("The correlation coefficient is: ")
print(corr_coeff)
print("-------------------------------------")
