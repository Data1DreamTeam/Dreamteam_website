def split_and_join_genres(urls):
    # Ensure the input is a string
    if isinstance(urls, str):
        # Strip any surrounding whitespace, split by '|', and process each genre
        genres = urls.strip().split('|')
        # For each genre, split by '/' and take the last part
        processed_genres = [genre.split('/')[-1] for genre in genres]
        # Join the processed genres with '|'
        return '|'.join(processed_genres)
    else:
        # Handle non-string cases (e.g., return empty string or another default value)
        return ''
